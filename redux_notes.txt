rtk query provides createApi

import {createApi} from "@reduxjs/toolit/query/react"
createApi function takes options names object which has baseQuey,endpoints
createApi({
baseQuery:fetchBaseQuery({baseUrl : "localhost/30000"}),
tagTypes:['tasks]
endpoints:(builder)=>({
getTasks:builder.query({
    query:()=>"/tasks",
    transformResponse:(tasks)=>tasks.reverse(), //provided by rtkquery
    provideTags:["tasks"],
}),
addTasks:builder.mutation({
    query:(task)=>({
        url:"/tasks",
        method:"POST",
        body:task,
    }),
    InvalidTags:["Tasks"],
    async onQueryStarted()
}),
}),
});
