//Difference between redux & redux toolkit
//1.redux toolkit provides configureStore() which simplifies store setup by automatically combining reducers, adding middleware,
// and enabling dev tools.
//2.createSlice() in redux toolkit simplifies reducer logic by allowing you to write "mutating" code that is safely 
//handled under the hood, while in redux you need to write immutable update logic.
//3.createAsyncThunk() in redux toolkit simplifies handling of async actions by automatically generating action 
//types and action creators for pending, fulfilled, and rejected states, while in redux you need to manually create these action 
//types and handle them in reducers.
//4.RTK Query is a powerful data fetching and caching tool built on top of redux toolkit that simplifies API interactions, while
// in redux you would need to set up your own data fetching logic and caching mechanisms.

//FETCHING APIS
//Rtk query is a powerful data fetching and caching tool built on top of redux toolkit that simplifies API interactions
//inside features/ create api/productApi.js
import { createApi, fetchBaseQuery } from '@reduxjs/toolkit/query/react';

export const productApi = createApi({
    reducerPath: 'productApi',
    baseQuery: fetchBaseQuery({ baseUrl: 'https://api.example.com/' }),
    tagTypes: ['Products'],
    endpoints: (builder) => ({
        getProducts: builder.query({
            query: () => 'products',
        }),
    }),
});

export const { useGetProductsQuery } = productApi;
